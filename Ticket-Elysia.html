  <script>

    // Récupérer les paramètres de l'URL (id et password)

    const urlParams = new URLSearchParams(window.location.search);

    const ticketId = urlParams.get('id');  // ID unique du ticket

    const password = urlParams.get('password');  // Mot de passe secret


    // Mot de passe secret pour les vigiles (à modifier)

    const correctPassword = 'secret2025';  // Ce mot de passe sera celui que tu utilises pour valider les tickets


    // Vérification de l'ID et du mot de passe

    if (ticketId && password === correctPassword) {

      // Si l'ID et le mot de passe sont valides, afficher un message de validation

      document.getElementById('message').innerHTML = `

        <p class="valid">Ticket validé ! Vous pouvez entrer.</p>

        <p>Bienvenue, votre accès est autorisé.</p>

      `;

      

      // Optionnel : Marquer le ticket comme utilisé (voir étape 3 pour Google Sheets)

      // markTicketAsUsed(ticketId);  // Appeler une fonction pour marquer ce ticket comme utilisé dans ta base de données (exemple avec Google Sheets)

      

    } else {

      // Si l'ID ou le mot de passe sont incorrects, afficher un message d'erreur

      document.getElementById('message').innerHTML = `

        <p class="invalid">QR Code invalide ou déjà utilisé.</p>

        <p>Veuillez vérifier votre code ou contactez l'organisateur.</p>

      `;

    }

  </script>
